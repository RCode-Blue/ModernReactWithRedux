Redux steps:

PART 1: Configure Reducers
--------------------------
1 Create the reducer (reducer_books.js)
2 Wire the reducer into the application

  (index.js):

  import BooksReducer from './reducer_books';

  const rootReducer = combineReducers({
   books: BooksReducer
  });

  export default rootReducer;


3 Render the information contained in the state on the screen
  book-list.js - convert from component to container:

  - In (book-list.js), add the following line:
    import { connect } from 'react-redux';

  - Move from "components" folder to "containers" folder
  
  - In (app.js), add the following line:
    import BookList from '../containers/book-list';


4 Get application state an turn it into props

  (book-list.js)
  function mapStateToProps(state){
      return{
          books: state.books
      };
  }
  

  export default connect(mapStateToProps)(BookList);


PART 2: CONFIGURE ACTIONS
-------------------------
5 Define Action Creators

(src/actions/index.js)
export function selectBook(book){
    console.log("a book: ", book.title);
}



6 Wire up the actions

(book-list.js)
import { selectBook } from '../actions/index';
import { bindActionCreators } from 'redux';

function mapDispatchToProps(dispatch) {
    return bindActionCreators({
        selectBook: selectBook                // key:value   selectBook was imported
    }, dispatch);
}

export default connect(mapStateToProps, mapDispatchToProps)(BookList);
